Number of Countries (Including Colonies) by Regime Category in 2020

count_regime_categories <- function(yr) {
  
  # Filter by the chosen year
  regime_data <- democracy_data |> 
    filter(year == yr, 
           !is.na(regime_category
                  )
           )|> 
    group_by(regime_category) |> 
    summarize(number_of_countries = n(), 
              .groups = "drop"
              ) |> 
  select(regime_category, number_of_countries)

  regime_data |>
    kable() |> 
    kable_styling() |>
    print()
  
  # Create a bar plot
  ggplot(regime_data, 
         aes(x = regime_category, 
             y = number_of_countries, 
             fill = regime_category
             )
         ) +
  geom_col() +
  labs(
    title = paste("Popularity by Regime in", yr),
    x = "Regime Category",
    y = "Number of Countries",
    fill = "Regime Category"
    ) + 
    theme_minimal()
  
}

Prevalence of Communism, Monarchies, and Democracies in 2020

count_political_ideologies <- function(selected_year) {
  
  # Filter by the selected year
  filtered_data <- democracy_data |> filter(year == selected_year)
  
  # Iterate over the ideology columns to count TRUE values
  ideology_cols <- c("is_communist", "is_democracy", "is_monarchy")
  ideology_cols
  
  counts <- map_int(ideology_cols, 
                    ~ sum(filtered_data[[.x]], 
                          na.rm = TRUE
                          )
                    )
  
  # Create a summary 
  summary_data <- tibble(
    political_ideology = c("Communism", "Democracy", "Monarchy"),
    number_of_countries = counts,
    year = selected_year
  ) |>
    group_by(political_ideology) |>
    summarize(number_of_countries = number_of_countries) 
  
  summary_data |> 
    kable() |> 
    kable_styling() |> 
    print()
  
  # Create a bar plot
  ggplot(summary_data, 
         aes(x = political_ideology, 
             y = number_of_countries, 
             fill = political_ideology
             )
         ) +
  geom_col() +
  labs(
    title = paste("Popularity of Communism, Democracies, and Monarchies in", selected_year),
    x = "Political Ideology",
    y = "Number of Countries",
    fill = "Political Ideology"
    ) + 
    theme_minimal()

}

Percent of Countries that have been Communist by Continent

percent_communist <- function(cont){
  democracy_data |>
    filter(continent == cont) |>
    distinct(country_name, is_communist) |>
    summarize(pct_communist = mean(is_communist, na.rm = TRUE)) |>
    pull(pct_communist)
}

percents <- democracy_data |>
  distinct(continent) |>
  pull(continent) |>
  map_dbl(percent_communist)

pct_table <- democracy_data |>
  distinct(continent) |>
  mutate(pct_communist = percents)
